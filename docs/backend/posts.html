# Posts

# Get posts

Returns a list of posts.

# URL

GET /api/groups/:groupId/posts

# Authorization

Bearer token

# Parameters

groupId required

page optional

Default value: 1

pageSize optional

Default value: 5

# Example Response

[
  {
    id: 1,
    name: "PostName",
    body: "PostBody",
    creationDate: "2022-12-22T18:09:22.511902Z",
  },
  {
    id: 2,
    name: "PostName",
    body: "PostBody",
    creationDate: "2022-12-22T18:09:44.434413Z",
  },
];

# Response codes

  • 400 - request was invalid or api failed to create user.
  • 200 - http request was successful


# Get post

Returns a post, specified by the postId parameter.

# URL

GET /api/groups/:groupId/posts/:postId

# Authorization

Bearer token

# Parameters

groupId required

postId required

# Example Response

{
    "id": 1,
    "name": "PostName",
    "body": "PostBody",
    "creationDate": "2022-12-22T18:09:22.511902Z"
}

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - http request was successful


# Create post

Creates a new post.

# URL

POST /api/groups/:groupId/posts

# Authorization

Bearer token

# Parameters

groupId required

# Example Request Body

{
    "Name":"PostName",
    "Body":"PostBody"
}

# Example Response

{
    "id": 1,
    "name": "PostName",
    "body": "PostBody",
    "creationDate": "2022-12-22T18:12:31.8752215Z"
}

# Response codes

  • 401 - not authorized
  • 201 - post created


# Update post

Updates a post, specified by the postId parameter.

# URL

PUT /api/groups/:groupId/posts/:postId

# Authorization

Bearer token

# Parameters

groupId required

postId required

# Example Request Body

{
    "Body":"PostBodyUpdated"
}

# Example Response

{
    "id": 1,
    "name": "PostName",
    "body": "PostBodyUpdated",
    "creationDate": "2022-12-22T18:09:22.511902Z"
}

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - post was updated


# Delete post

Deletes a post, specified by the postId parameter.

# URL

DELETE /api/groups/:groupId/posts/:postId

# Authorization

Bearer token

# Parameters

groupId required

postId required

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 204 - post deleted
