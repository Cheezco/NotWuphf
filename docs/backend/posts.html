# Posts

# Get posts

Returns a list of posts.

# URL

GET /api/groups/:groupId/posts

# Parameters

groupId required

page optional

Default value: 1

pageSize optional

Default value: 5

# Response

[
  {
    id: 0,
    name: "",
    body: "",
    creationDate: "",
  },
];


# Get post

Returns a post, specified by the postId parameter.

# URL

GET /api/groups/:groupId/posts/:postId

# Parameters

groupId required

postId required

# Response

{
    id: 0,
    name: "",
    body: "",
    creationDate: "",
};
