# Comments

# Get comments

Returns a list of comments.

# URL

GET /api/groups/:groupId/posts/:postId/comments

# Authorization

Bearer token

# Parameters

groupId required

postId required

page optional

Default value: 1

pageSize optional

Default value: 5

# Example Response

[
  {
    id: 1,
    content: "CommentContent",
    creationDate: "2022-12-22T18:18:05.946582Z",
  },
  {
    id: 2,
    content: "CommentContent",
    creationDate: "2022-12-22T18:18:09.925098Z",
  },
];

# Example Response codes

  • 401 - not authorized
  • 200 - http request was successful


# Get comment

Returns a comment, specified by the commentId parameter.

# URL

GET /api/groups/:groupId/posts/:postId/comments/:commentId

# Authorization

Bearer token

# Parameters

groupId required

postId required

commentId required

# Example Response

{
    "id": 1,
    "content": "CommentContent",
    "creationDate": "2022-12-22T18:18:05.946582Z"
}

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - http request was successful


# Create comment

Creates a new comment.

# URL

POST /api/groups/:groupId/posts/:postId/comments

# Authorization

Bearer token

# Parameters

groupId required

postId required

# Example Request Body

{
    "Content":"CommentContent"
}

# Example Response

{
    "id": 1,
    "content": "CommentContent",
    "creationDate": "2022-12-22T18:18:09.9250982Z"
}

# Response codes

  • 401 - not authorized
  • 201 - comment created


# Update comment

Updates a comment, specified by the commentId parameter.

# URL

PUT /api/groups/:groupId/posts/:postId/comments/:commentId

# Authorization

Bearer token

# Parameters

groupId required

postId required

commentId required

# Example Request Body

{
    "Content":"UpdatedCommentContent"
}

# Example Response

{
    "id": 1,
    "content": "UpdatedCommentContent",
    "creationDate": "2022-12-22T18:18:05.946582Z"
}

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - comment was updated


# Delete comment

Deletes a comment, specified by the commentId parameter.

# URL

DELETE /api/groups/:groupId/posts/:postId/comments/:commentId

# Authorization

Bearer token

# Parameters

groupId required

postId required

commentId required

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 204 - comment deleted
