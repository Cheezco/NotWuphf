# Groups

# Get groups

Returns a list of groups.

# URL

GET /api/groups

# Authorization

Bearer token

# Parameters

page optional

Default value: 1

pageSize optional

Default value: 5

# Example Response

[
  {
    id: 1,
    name: "1234",
    description: "124",
    visibility: 0,
    creationDate: "2022-12-16T12:04:27.697862Z",
  },
  {
    id: 2,
    name: "777",
    description: "5555",
    visibility: 0,
    creationDate: "2022-12-16T12:04:22.573302Z",
  },
  {
    id: 3,
    name: "GroupName",
    description: "GroupDescription",
    visibility: 0,
    creationDate: "2022-12-22T17:56:11.800531Z",
  },
  {
    id: 4,
    name: "GroupName",
    description: "GroupDescription",
    visibility: 0,
    creationDate: "2022-12-22T17:56:20.433403Z",
  },
];

# Response codes

  • 401 - not authorized
  • 200 - http request was successful


# Get group

Returns a group, specified by the groupId parameter.

# URL

GET /api/groups/:groupId

# Authorization

Bearer token

# Parameters

groupId required

# Example Response

{
    "id": 1,
    "name": "GroupName",
    "description": "GroupDescription",
    "visibility": 0,
    "creationDate": "2022-12-22T17:56:11.800531Z"
}

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - http request was successful


# Create group

Creates a new group.

# URL

POST /api/groups/:groupId

# Authorization

Bearer token

# Example Request Body

{
    "Name":"GroupName",
    "Description":"GroupDescription",
    "GroupVisibility":0
}

# Example Response

{
    id: 0,
    name: "",
    description: "",
    visibility: "",
    creationDate: "",
};

# Response codes

  • 401 - not authorized
  • 201 - group created


# Update group

Updates a group, specified by the groupId parameter.

# URL

PUT /api/groups/:groupId

# Authorization

Bearer token

# Parameters

groupId required

# Example Request Body

{
    "Name":"GroupNameUpdated777",
    "Description":"GroupDescriptionUpdated",
    "GroupVisibility":0
}

# Example Response

{
    id: 0,
    name: "",
    description: "",
    visibility: "",
    creationDate: "",
};

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 200 - group was updated


# Delete group

Deletes a group, specified by the groupId parameter.

# URL

DELETE /api/groups/:groupId

# Authorization

Bearer token

# Parameters

groupId required

# Response codes

  • 401 - not authorized
  • 404 - not found
  • 204 - group deleted
